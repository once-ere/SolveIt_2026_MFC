The :+ is a form of parameter expansion:

${parameter:+[word]} : Use Alternative Value.

If parameter is unset or null, null shall be substituted; otherwise, the expansion of word (or an empty string if word is omitted) shall be substituted.

In other words, if the variable $var is defined, echo ${var:+foo} will print foo and, if it is not, it will print the empty string.

The second : is nothing special. It is the character used as a separator in the list of directories in $PATH. So, PATH="/usr/local/bin:/usr/bin${PATH:+:${PATH}}" is a shorthand way of writing:

if [ -z "$PATH" ]; then
    PATH=/usr/local/bin:/usr/bin
else
    PATH=/usr/local/bin:/usr/bin:$PATH
fi
It's just a clever trick to avoid adding an extra : when $PATH is not set. For example:

$ PATH="/usr/bin"
$ PATH="/new/dir:$PATH" ## Add a directory
$ echo "$PATH"
/new/dir:/usr/bin
But if PATH is unset:

$ unset PATH
$ PATH="/new/dir:$PATH"
$ echo "$PATH"
/new/dir:
A : by itself adds the current directory to the $PATH. Using PATH="/new/dir${PATH:+:$PATH}" avoids this. So sure, you can use PATH="${PATH:+${PATH}:}/usr/local/bin:/usr/bin" if you want to, or you can use PATH="$PATH:/usr/local/bin:/usr/bin" if you prefer. The only difference is that the former might add an extra :, thereby adding your current directory to your $PATH.

// src/eval.rs
use crate::ast::{BinOp, Expr, Stmt};
use crate::symbol::{SymbolTable, Symbol, SymbolError};
use thiserror::Error;

#[derive(Error, Debug)]
pub enum EvalError {
    #[error("Symbol error: {0}")]
    Symbol(#[from] SymbolError),
    #[error("Math error: {0}")]
    Math(String),
}

pub struct Evaluator {
    symbols: SymbolTable,
}

impl Evaluator {
    pub fn new(symbols: SymbolTable) -> Self {
        Self { symbols }
    }

    pub fn eval(&mut self, stmts: &[Stmt]) -> Result<f64, EvalError> {
        let mut result = 0.0;
        for stmt in stmts {
            result = self.eval_stmt(stmt)?;
        }
        Ok(result)
    }

    fn eval_stmt(&mut self, stmt: &Stmt) -> Result<f64, EvalError> {
        match stmt {
            Stmt::Expr(expr) => self.eval_expr(expr),
            Stmt::FunctionDef(name, args, body) => {
                self.symbols.define_function(name.clone(), args.clone(), body.clone())?;
                Ok(0.0)
            }
        }
    }

    fn eval_expr(&mut self, expr: &Expr) -> Result<f64, EvalError> {
        match expr {
            Expr::Number(n) => Ok(*n),
            Expr::Variable(name) => self.symbols.lookup(name).map(|s| s.value()),
            Expr::Assignment(name, expr) => {
                let val = self.eval_expr(expr)?;
                self.symbols.define_variable(name.clone(), val)?;
                Ok(val)
            }
            Expr::BinaryOp(lhs, op, rhs) => {
                let lhs_val = self.eval_expr(lhs)?;
                let rhs_val = self.eval_expr(rhs)?;
                match op {
                    BinOp::Add => Ok(lhs_val + rhs_val),
                    BinOp::Subtract => Ok(lhs_val - rhs_val),
                    BinOp::Multiply => Ok(lhs_val * rhs_val),
                    BinOp::Divide => Ok(lhs_val / rhs_val),
                    BinOp::Power => Ok(libm::pow(lhs_val, rhs_val)),
                }
            }
            Expr::FunctionCall(name, args) => {
                let arg_vals = args.iter()
                    .map(|a| self.eval_expr(a))
                    .collect::<Result<Vec<_>, _>>()?;
                self.symbols.call_function(name, &arg_vals)
            }
            Expr::Block(exprs) => {
                let mut result = 0.0;
                for expr in exprs {
                    result = self.eval_expr(expr)?;
                }
                Ok(result)
            }
        }
    }
}

// src/lexer.rs
use regex::Regex;
use std::iter::Peekable;
use std::str::Chars;

#[derive(Debug, Clone, PartialEq)]
pub enum Token {
    Number(f64),
    Ident(String),
    Plus,
    Minus,
    Star,
    Slash,
    Caret,
    Equal,
    LParen,
    RParen,
    LBrace,
    RBrace,
    Semicolon,
    EOF,
}

pub struct Lexer<'a> {
    chars: Peekable<Chars<'a>>,
}

impl<'a> Lexer<'a> {
    pub fn new(input: &'a str) -> Self {
        Lexer {
            chars: input.chars().peekable(),
        }
    }

    fn skip_whitespace(&mut self) {
        while let Some(c) = self.chars.peek() {
            if c.is_whitespace() {
                self.chars.next();
            } else {
                break;
            }
        }
    }

    fn read_number(&mut self, first: char) -> f64 {
        let mut num = String::new();
        num.push(first);

        while let Some(c) = self.chars.peek() {
            if c.is_numeric() || *c == '.' {
                num.push(*c);
                self.chars.next();
            } else {
                break;
            }
        }

        num.parse().unwrap_or_else(|_| panic!("Invalid number: {}", num))
    }

    fn read_ident(&mut self, first: char) -> String {
        let mut ident = String::new();
        ident.push(first);

        while let Some(c) = self.chars.peek() {
            if c.is_alphanumeric() || *c == '_' {
                ident.push(*c);
                self.chars.next();
            } else {
                break;
            }
        }

        ident
    }
}

impl<'a> Iterator for Lexer<'a> {
    type Item = Token;

    fn next(&mut self) -> Option<Self::Item> {
        self.skip_whitespace();

        match self.chars.next()? {
            '+' => Some(Token::Plus),
            '-' => Some(Token::Minus),
            '*' => Some(Token::Star),
            '/' => Some(Token::Slash),
            '^' => Some(Token::Caret),
            '=' => Some(Token::Equal),
            '(' => Some(Token::LParen),
            ')' => Some(Token::RParen),
            '{' => Some(Token::LBrace),
            '}' => Some(Token::RBrace),
            ';' => Some(Token::Semicolon),
            c if c.is_numeric() => Some(Token::Number(self.read_number(c))),
            c if c.is_alphabetic() => Some(Token::Ident(self.read_ident(c))),
            _ => None,
        }
    }
}

use std::collections::HashMap;
use crate::ast::Expr;
use thiserror::Error;

#[derive(Debug, Clone)]
pub enum Symbol {
    Variable(f64),
    Function(Vec<String>, Expr),
    Builtin(fn(&[f64]) -> Result<f64, String>),
}

impl Symbol {
    pub fn value(&self) -> f64 {
        match self {
            Symbol::Variable(val) => *val,
            _ => panic!("Cannot get value from function"),
        }
    }
}

#[derive(Error, Debug)]
pub enum SymbolError {
    #[error("Undefined symbol: {0}")]
    Undefined(String),
    #[error("Invalid arguments for function {0}")]
    InvalidArguments(String),
}

pub struct SymbolTable {
    symbols: HashMap<String, Symbol>,
}

impl SymbolTable {
    pub fn new() -> Self {
        Self {
            symbols: HashMap::new(),
        }
    }

    pub fn init_builtins(&mut self) {
        self.define_builtin("sqrt", 1, |args| {
            Ok(libm::sqrt(args[0]))
        });
        self.define_builtin("exp", 1, |args| {
            Ok(libm::exp(args[0]))
        });
        self.define_variable("PI", std::f64::consts::PI).unwrap();
        self.define_variable("E", std::f64::consts::E).unwrap();
    }

    pub fn lookup(&self, name: &str) -> Result<Symbol, SymbolError> {
        self.symbols.get(name)
            .cloned()
            .ok_or_else(|| SymbolError::Undefined(name.to_string()))
    }

    pub fn define_variable(&mut self, name: String, value: f64) -> Result<(), SymbolError> {
        self.symbols.insert(name, Symbol::Variable(value));
        Ok(())
    }

    pub fn define_function(
        &mut self,
        name: String,
        params: Vec<String>,
        body: Expr,
    ) -> Result<(), SymbolError> {
        self.symbols.insert(name, Symbol::Function(params, body));
        Ok(())
    }

    pub fn define_builtin(
        &mut self,
        name: &str,
        arity: usize,
        func: fn(&[f64]) -> Result<f64, String>,
    ) {
        self.symbols.insert(
            name.to_string(),
            Symbol::Builtin(|args| {
                if args.len() != arity {
                    Err(format!("Expected {} arguments", arity))
                } else {
                    func(args)
                }
            }),
        );
    }

    pub fn call_function(
        &self,
        name: &str,
        args: &[f64],
    ) -> Result<f64, SymbolError> {
        match self.lookup(name)? {
            Symbol::Builtin(func) => func(args)
                .map_err(|e| SymbolError::InvalidArguments(e)),
            Symbol::Function(params, body) => {
                // Implementation would need to handle function evaluation
                Err(SymbolError::InvalidArguments("Not implemented".to_string()))
            }
            _ => Err(SymbolError::InvalidArguments(name.to_string())),
        }
    }
}

    /// Create new evaluator with symbol table
    pub fn new(symbols: SymbolTable) -> Self {
        Self { symbols }

    /// Evaluate a list of statements
    pub fn eval(&mut self, stmts: &[Stmt]) -> Result<f64> {
        let mut result = 0.0;
        for stmt in stmts {
            result = self.eval_stmt(stmt)
                .context(format!("Failed evaluating statement: {:?}", stmt))?;
        }
        Ok(result)

    fn eval_stmt(&mut self, stmt: &Stmt) -> Result<f64> {
        match stmt {
            Stmt::Expr(expr) => self.eval_expr(expr),
            Stmt::FunctionDef(name, args, body) => {
                self.symbols.define_function(name.clone(), args.clone(), body.clone())?;
                Ok(0.0) // Function definition doesn't return value
            }
        }
    }

    fn eval_expr(&mut self, expr: &Expr) -> Result<f64> {
        match expr {
            Expr::Number(n) => Ok(*n),
            Expr::Variable(name) => self.symbols
                .lookup(name)
                .map(|s| s.value())
                .context(format!("Undefined variable: {}", name)),

            Expr::Assignment(name, expr) => {
                let val = self.eval_expr(expr)?;
                self.symbols.define_variable(name.clone(), val)
                    .context(format!("Failed assigning variable: {}", name))?;
                Ok(val)
            }

            Expr::BinaryOp(lhs, op, rhs) => {
                let lhs_val = self.eval_expr(lhs)?;
                let rhs_val = self.eval_expr(rhs)?;
                match op {
                    BinOp::Add => Ok(lhs_val + rhs_val),
                    BinOp::Subtract => Ok(lhs_val - rhs_val),
                    BinOp::Multiply => Ok(lhs_val * rhs_val),
                    BinOp::Divide => {
                        if rhs_val == 0.0 {
                            Err(SymbolError::MathError("Division by zero".into()))?;
                        }
                        Ok(lhs_val / rhs_val)
                    }
                    BinOp::Power => Ok(libm::pow(lhs_val, rhs_val)),
                }
            }

            Expr::FunctionCall(name, args) => {
                let arg_vals = args.iter()
                    .map(|a| self.eval_expr(a))
                    .collect::<Result<Vec<_>>>()?;
                self.symbols.call_function(name, &arg_vals)
                    .context(format!("Function call failed: {}", name))
            }

            Expr::Block(exprs) => {
                let mut result = 0.0;
                for expr in exprs {
                    result = self.eval_expr(expr)?;
                }
                Ok(result)
            }
        }

    Equal,      // =

/// Lexer converts source code into tokens
pub struct Lexer<'a> {
    chars: Peekable<Chars<'a>>,
}

impl<'a> Lexer<'a> {
    /// Create new lexer from input string
    pub fn new(input: &'a str) -> Self {
        Lexer {
            chars: input.chars().peekable(),
        }
    }

    /// Skip whitespace characters
    fn skip_whitespace(&mut self) {
        while let Some(c) = self.chars.peek() {
            if c.is_whitespace() {
                self.chars.next();
            } else {
                break;
            }
        }
    }

    /// Read a number literal (supports integers and floats)
    fn read_number(&mut self, first: char) -> f64 {
        let mut num = String::new();
        num.push(first);

        while let Some(c) = self.chars.peek() {
            if c.is_numeric() || *c == '.' {
                num.push(*c);
                self.chars.next();
            } else {
                break;
            }
        }

        num.parse().unwrap_or_else(|_| panic!("Invalid number: {}", num))
    }

    /// Read an identifier (variables/functions)
    fn read_ident(&mut self, first: char) -> String {
        let mut ident = String::new();
        ident.push(first);

        while let Some(c) = self.chars.peek() {
            if c.is_alphanumeric() || *c == '_' {
                ident.push(*c);
                self.chars.next();
            } else {
                break;
            }
        }

        ident
    }
}

impl<'a> Iterator for Lexer<'a> {
    type Item = Token;

    /// Generate next token from input
    fn next(&mut self) -> Option<Self::Item> {
        self.skip_whitespace();

        match self.chars.next()? {
            '+' => Some(Token::Plus),
            '-' => Some(Token::Minus),
            '*' => Some(Token::Star),
            '/' => Some(Token::Slash),
            '^' => Some(Token::Caret),
            '=' => Some(Token::Equal),
            '(' => Some(Token::LParen),
            ')' => Some(Token::RParen),
            '{' => Some(Token::LBrace),
            '}' => Some(Token::RBrace),
            ';' => Some(Token::Semicolon),
            c if c.is_numeric() => Some(Token::Number(self.read_number(c))),
            c if c.is_alphabetic() => Some(Token::Ident(self.read_ident(c))),
            _ => None,
        }
    }

If parameter is unset or null, null shall be substituted; otherwise, the expansion of word (or an empty string if word is omitted) shall be substituted.

In other words, if the variable $var is defined, echo ${var:+foo} will print foo and, if it is not, it will print the empty string.

The second : is nothing special. It is the character used as a separator in the list of directories in $PATH. So, PATH="/usr/local/bin:/usr/bin${PATH:+:${PATH}}" is a shorthand way of writing:

if [ -z "$PATH" ]; then
    PATH=/usr/local/bin:/usr/bin
else
    PATH=/usr/local/bin:/usr/bin:$PATH
fi
It's just a clever trick to avoid adding an extra : when $PATH is not set. For example:

$ PATH="/usr/bin"
$ PATH="/new/dir:$PATH" ## Add a directory
$ echo "$PATH"
/new/dir:/usr/bin
But if PATH is unset:

$ unset PATH
$ PATH="/new/dir:$PATH"
$ echo "$PATH"
/new/dir:
A : by itself adds the current directory to the $PATH. Using PATH="/new/dir${PATH:+:$PATH}" avoids this. So sure, you can use PATH="${PATH:+${PATH}:}/usr/local/bin:/usr/bin" if you want to, or you can use PATH="$PATH:/usr/local/bin:/usr/bin" if you prefer. The only difference is that the former might add an extra :, thereby adding your current directory to your $PATH.
